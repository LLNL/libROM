86 ScalaMat::ScalaMat(LocalMatrix &&other,
87  std::shared_ptr<const Context> ctxt) : m_globm(other.m()), m_globn(other.n()),
88  m_mb(std::max(other.m(), other.n())),
89  m_nb(m_mb), m_ctxt(ctxt), m_own_data(other.own_data())
90 {
91  // Require that other is column major.
92  if (other.ordering() == ROW_MAJOR)
93  {
94  if (mpi_rank() == 0)
95  {
96  std::cerr << "ScalaMat(LocalMatrix): Attempted to create a ScalaMat from a row major local\n";
97  std::cerr << "(in " << __func__ << " at " << __FILE__ << ':'
98  << __LINE__ << ")\n";
99  }
100  throw ScalaMatException("It is not allowed to construct a ScalaMat from a row major matrix");
101  }
102 
103  int nprow, npcol, pi = 0, pj = 0;
104 
105  if (ctxt)
106  {
107  blacs_gridinfo_wrapper(ctxt->id(), &nprow, &npcol, &pi, &pj);
108  }
109 
110  int coords[] = {pi, pj};
111 
112  if (ctxt == nullptr || nprow == -1)
113  { // We are not in the context.
114  if (mpi_rank() == other.srcrank()) {
115  std::cerr << "ScalaMat(LocalMatrix): Attempt to construct from a local matrix\n"
116  << "on a process outside the context\n";
117  throw ScalaMatException("Local matrix not in context");
118  }
119  m_mb = m_nb = 0;
120  m_rowsrc = m_colsrc = 0;
121  m_ctxt = nullptr;
122  m_data = new double[0];
123  m_own_data = true;
124  MPI_Bcast(coords, 2, MPI_INT, other.srcrank(), MPI_COMM_WORLD);
125  return;
126  }
127 
128  // Only processes in the participating context are still executing.
129  // Set the coordinates of the source process in the grid.
130  MPI_Bcast(coords, 2, MPI_INT, other.srcrank(), MPI_COMM_WORLD);
131  m_rowsrc = coords[0];
132  m_colsrc = coords[1];
133 
134  if (mpi_rank() == other.srcrank())
135  {
136  m_data = other.release_data();
137  }
138  else
139  {
140  m_data = new double[0];
141  m_own_data = true;
142  }
143 }

316 LocalMatrix &LocalMatrix::operator=(const ScalaMat::Submatrix &other)
317 {
318  std::array<int, 9> dst_desc, src_desc;
319  bool owned = m_own_data;
320  m_own_data = false;
321 
322  switch (m_ordering)
323  {
324  case ROW_MAJOR:
325  {
326  m_ordering = COL_MAJOR;
327  std::swap(m_m, m_n);
328  bool need_copy = false;
329  if (mpi_rank() == m_rank)
330  {
331  if (!other.parent().context())
332  {
333  need_copy = true;
334  }
335  }
336  MPI_Bcast(&need_copy, 1, MPI_C_BOOL, m_rank, MPI_COMM_WORLD);
337 
338  if (!need_copy)
339  {
340  ScalaMat target(std::move(*this), other.parent().context());
341 
342  target.initialize_descriptor(dst_desc);
343  other.parent().initialize_descriptor(src_desc);
344  pdgeadd_wrapper('T', target.m(), target.n(), 1.0,
345  other.parent().data(), other.i(), other.j(),
346  src_desc.data(), 0.0, target.data(), 1, 1,
347  dst_desc.data());
348  }
349  else
350  {
351  ScalaMat tmp(m_n, m_m);
352  tmp(rowrange(1, m_n), colrange(1, m_m)) = other;
353  ScalaMat target(std::move(*this));
354  tmp.transpose(target);
355  }
356 
357  std::swap(m_m, m_n);
358  m_ordering = ROW_MAJOR;
359  break;
360  }
361  default:
362  {
363  ScalaMat target(std::move(*this));
364  target(rowrange(1, target.m()), colrange(1, target.n())) = other;
365  break;
366  }
367  }
368 
369  m_own_data = owned;
370  return *this;
371 }
