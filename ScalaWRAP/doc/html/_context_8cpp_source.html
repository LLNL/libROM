37 std::pair<int, int> best_distribution(int nprocs)
38 {
39  int i = std::floor(std::sqrt(nprocs));
40  while (nprocs % i != 0) {
41  i -= 1;
42  }
43  return std::make_pair(i, nprocs / i);
44 }

69 Context::Context(const std::vector<int>& process_map, int nprow)
70 {
71  int mrank, nprocs;
72  std::tie(nprocs, mrank) = mpi_global_params();
73 
74  assert(nprocs >= 1);
75  if (process_map.size() > static_cast<unsigned>(nprocs)) {
76  if (mrank == 0) {
77  std::cerr << "Error: process_map.size() = " << process_map.size()
78  << " but size of MPI_COMM_WORLD is only " << nprocs << '\n';
79  std::cerr << "(in " << __func__ << " at " << __FILE__ << ':'
80  << __LINE__ << '\n';
81  }
82  throw ContextException("Context process_map constructor: Requested unfulfillable grid");
83  }
84 
85  // Check for multiple ranks in process_map, or processes outside of the
86  // valid index range.
87  std::set<int> rankset;
88  for (auto rank: process_map) {
89  if (rankset.count(rank) == 1) {
90  if (mrank == 0) {
91  std::cerr <<
92  "Error: Multiple entries in process_map refer to the same rank\n"
93  << "(in " << __func__ << " at " << __FILE__ << ':'
94  << __LINE__ << '\n';
95  }
96  throw ContextException("Context process_map constructor: multiple specs of same rank");
97  } else if (rank >= nprocs) {
98  if (mrank == 0) {
99  std::cerr <<
100  "Error: rank = " << rank << " is outside of the index "
101  << "range for processes (nprocs = " << nprocs << ")\n";
102  std::cerr << "(in " << __func__ << " at " << __FILE__ << ':'
103  << __LINE__ << '\n';
104  }
105  throw ContextException("Context process_map constructor: rank outside the valid range");
106  }
107  rankset.insert(rank);
108  }
109 
110  if (process_map.size() % nprow != 0) {
111  if (mrank == 0) {
112  std::cerr <<
113  "Error: process_map.size() is not a multiple of nprow\n"
114  << "(in " << __func__ << " at " << __FILE__ << ':' << __LINE__
115  << '\n';
116  }
117  throw ContextException("Context process_map constructor: process_map.size % nprow != 0");
118  }
119 
120  m_contextid = blacs_gridmap_wrapper(process_map.data(), nprow, nprow,
121  process_map.size() / nprow);
122 }
