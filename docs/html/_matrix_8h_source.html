145  void
146  setSize(
147  int num_rows,

149  {
150  int new_size = num_rows*num_cols;
151  if (new_size > d_alloc_size) {
152  if (!d_owns_data) {
153  CAROM_ERROR("Can not reallocate externally owned storage.");
154  }
155  if (d_mat) {

158  d_mat = new double [new_size] {};
159  d_alloc_size = new_size;
160  }
161  d_num_rows = num_rows;
162  d_num_cols = num_cols;
163  if (d_distributed) {
164  calculateNumDistributedRows();
165  }
166  }
