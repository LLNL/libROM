238  void BuildSampleMesh(ParMesh& pmesh, ParFiniteElementSpace& H1DummySpace,
239  ParFiniteElementSpace& fespace1, ParFiniteElementSpace& fespace2,
240  const set<int>& elems, Mesh*& sample_mesh, vector<int>& stencil_dofs,
241  vector<int>& elemLocalIndices, vector<map<int, int> >& elemLocalIndicesInverse)
242  {
243  // Get the number of stencil elements coming from each processor and the
244  // total number of stencil elements.
245  int num_procs = -1;
246  MPI_Comm_size(MPI_COMM_WORLD, &num_procs);
247 
248  vector<int> procNumElems(num_procs);
249 
250  const int local_num_elems = elems.size();
251  int* cts = new int [num_procs];
252  int* offsets = new int [num_procs];
253  MPI_Allgather(&local_num_elems, 1, MPI_INT, cts, 1, MPI_INT, MPI_COMM_WORLD);
254  int sample_mesh_num_elems = 0;
255  offsets[0] = 0;
256  for (int i = 0; i < num_procs; ++i) {
257  sample_mesh_num_elems += cts[i];
258  procNumElems[i] = cts[i];
259  cts[i] *= 2;
260  if (i > 0)
261  offsets[i] = offsets[i-1] + cts[i-1];
262  }
263 
264  #ifdef FULL_DOF_STENCIL
265  const int N1 = fespace1.GetVSize();
266  #else
267  const int N1 = fespace1.TrueVSize();
268  #endif
269  const int d = pmesh.Dimension();
270 
271  // Each processor generates for each of its stencil elements a list of global
272  // TDofs (which are equivalent to vertices), a list of element attributes,
273  // and a list of vertices (coords). These get communicated to P0. For each
274  // stencil mesh element, P0 goes through the list of vertices and constructs
275  // a map<int, int> mapping these global TDofs to local vertex ids. For each
276  // unique TDof it finds it adds a vertex to the sample mesh. Then it adds
277  // each element, attribute, and connectivity.
278  vector<int> my_element_attr(2*local_num_elems); // Stores element attribute and local index
279  set<int> element_dofs;
280  int attr_idx = 0;
281  int conn_idx = 0;
282  int coords_idx = 0;
283  Array<int> elVert;
284  pmesh.GetElementVertices(*elems.begin(), elVert);
285  int numElVert = elVert.Size(); // number of vertices per element
286  MFEM_VERIFY(numElVert > 0, "");
287 
288  vector<int> my_element_vgid(numElVert*local_num_elems); // vertex global indices, for each element
289  vector<double> my_element_coords(d*numElVert*local_num_elems);
290 
291  for (set<int>::iterator it = elems.begin(); it != elems.end(); ++it) {
292  const int elId = *it;
293  InsertElementDofs(fespace1, elId, 0, element_dofs);
294  InsertElementDofs(fespace2, elId, N1, element_dofs);
295 
296  pmesh.GetElementVertices(elId, elVert);
297  MFEM_VERIFY(numElVert == elVert.Size(), ""); // Assuming a uniform element type in the mesh.
298  // NOTE: to be very careful, it should be verified that this is the same across all processes.
299 
300  Array<int> dofs;
301  H1DummySpace.GetElementDofs(elId, dofs);
302  MFEM_VERIFY(numElVert == dofs.Size(), ""); // Assuming a bijection between vertices and H1 dummy space DOF's.
303 
304  for (int i = 0; i < numElVert; ++i) {
305  my_element_vgid[conn_idx++] = H1DummySpace.GetGlobalTDofNumber(dofs[i]);
306  double* coords = pmesh.GetVertex(elVert[i]);
307  for (int j=0; j<d; ++j)
308  my_element_coords[coords_idx++] = coords[j];
309  }
310 
311  my_element_attr[attr_idx++] = pmesh.GetAttribute(elId);
312  my_element_attr[attr_idx++] = elId;
313  }
314 
315  stencil_dofs.assign(element_dofs.begin(), element_dofs.end());
316 
317  #ifdef FULL_DOF_STENCIL
318  AugmentDofListWithOwnedDofs(stencil_dofs, fespace1, fespace2);
319  #endif
320 
321  MFEM_VERIFY(coords_idx == d*numElVert*local_num_elems, "");
322  MFEM_VERIFY(conn_idx == numElVert*local_num_elems, "");
323  MFEM_VERIFY(attr_idx == 2*local_num_elems, "");
324 
325  // Gather all the element attributes from all processors.
326 
327  vector<int> element_attr(2*sample_mesh_num_elems);
328  MPI_Allgatherv(&my_element_attr[0], 2*local_num_elems, MPI_INT,
329  &element_attr[0], cts, offsets, MPI_INT, MPI_COMM_WORLD);
330 
331  // Gather all the element connectivities from all processors.
332  offsets[0] = 0;
333  cts[0] = numElVert*cts[0]/2;
334  for (int i = 1; i < num_procs; ++i) {
335  cts[i] = numElVert*cts[i]/2;
336  offsets[i] = offsets[i-1] + cts[i-1];
337  }
338  vector<int> element_vgid(numElVert*sample_mesh_num_elems);
339  MPI_Allgatherv(&my_element_vgid[0], numElVert*local_num_elems, MPI_INT,
340  &element_vgid[0], cts, offsets, MPI_INT, MPI_COMM_WORLD);
341 
342  // Gather all the element coordinates from all processors.
343  offsets[0] = 0;
344  cts[0] = d*cts[0];
345  for (int i = 1; i < num_procs; ++i) {
346  cts[i] = d*cts[i];
347  offsets[i] = offsets[i-1] + cts[i-1];
348  }
349  vector<double> element_coords(d*numElVert*sample_mesh_num_elems);
350  MPI_Allgatherv(&my_element_coords[0], d*numElVert*local_num_elems, MPI_DOUBLE,
351  &element_coords[0], cts, offsets, MPI_DOUBLE, MPI_COMM_WORLD);
352  delete [] cts;
353  delete [] offsets;
354 
355  // element_vgid holds vertices as global ids. Vertices may be shared
356  // between elements so we don't know the number of unique vertices in the
357  // sample mesh. Find all the unique vertices and construct the map of
358  // global dof ids to local dof ids (vertices). Keep track of the number of
359  // unique vertices.
360  set<int> unique_gdofs;
361  map<int, int> unique_gdofs_first_appearance;
362  for (int i = 0; i < numElVert*sample_mesh_num_elems; ++i) {
363  int gdof = element_vgid[i];
364  if (unique_gdofs.insert(gdof).second) {
365  unique_gdofs_first_appearance.insert(make_pair(gdof, i));
366  }
367  }
368  int sample_mesh_num_verts = unique_gdofs.size();
369  map<int, int> unique_gdofs_2_vertex;
370  map<int, int> vertex_2_unique_gdofs;
371  int idx = 0;
372  for (set<int>::iterator it = unique_gdofs.begin();
373  it != unique_gdofs.end(); ++it) {
374  unique_gdofs_2_vertex.insert(make_pair(*it, idx));
375  vertex_2_unique_gdofs.insert(make_pair(idx, *it));
376  ++idx;
377  }
378 
379  // Now we have enough info to build the sample mesh.
380  printf("sample mesh has %d elements, %d vertices\n",
381  sample_mesh_num_elems, sample_mesh_num_verts);
382 
383  sample_mesh = new Mesh(d, sample_mesh_num_verts, sample_mesh_num_elems);
384 
385  // For each vertex that we found, add its coordinates to the stencil mesh.
386  for (int vert = 0; vert < sample_mesh_num_verts; ++vert) {
387  int unique_gdof = vertex_2_unique_gdofs[vert];
388  int first_conn_ref = unique_gdofs_first_appearance[unique_gdof];
389  int coord_loc = d*first_conn_ref;
390  sample_mesh->AddVertex(&element_coords[coord_loc]);
391  }
392 
393  // Now add each element and give it its attributes and connectivity.
394  const int elGeom = pmesh.GetElementBaseGeometry(0);
395  idx = 0;
396  elemLocalIndices.resize(sample_mesh_num_elems);
397  elemLocalIndicesInverse.resize(num_procs);
398  int ielem = 0;
399  for (int p=0; p<num_procs; ++p)
400  {
401  for (int i=0; i<procNumElems[p]; ++i, ++ielem)
402  {
403  Element* sel = sample_mesh->NewElement(elGeom);
404  sel->SetAttribute(element_attr[2*ielem]);
405 
406  elemLocalIndices[ielem] = element_attr[(2*ielem)+1];
407  elemLocalIndicesInverse[p][elemLocalIndices[ielem]] = ielem;
408 
409  Array<int> sv(numElVert);
410  for (int vert = 0; vert < numElVert; ++vert) {
411  sv[vert] = unique_gdofs_2_vertex[element_vgid[idx++]];
412  }
413  sel->SetVertices(sv);
414 
415  sample_mesh->AddElement(sel);
416  }
417  }
418 
419  MFEM_VERIFY(ielem == sample_mesh_num_elems, "");
420 
421  sample_mesh->FinalizeTopology();
422  }

471  void Set_s2sp(const int myid, const int num_procs, const int spN1, const int global_num_sample_dofs, const vector<int>& local_num_sample_dofs,
472  const vector<vector<int> >& local_num_sample_dofs_sub, const vector<vector<int> >& localSampleDofsToElem_sub,
473  const vector<vector<int> >& localSampleDofsToElemDof_sub,
474  const vector<vector<int> >& sample_dofs_sub_to_sample_dofs,
475  const vector<map<int, int> >& elemLocalIndicesInverse,
476  ParFiniteElementSpace** spfespace,
477  vector<int>& s2sp)
478  {
479  // Set s2sp, which is a map from the selected indices in sample_dofs for all processes to DOF's in sample_mesh.
480  // This is done by the following steps:
481  // 1. For each local sample DOF, determine its fespace and find an element to which it belongs and its local index with respect to the element DOF ordering.
482  // 2. Communicate this data (one process-local element index and an element DOF index) from each process to P0.
483  // 3. On P0, loop over each process's sample DOF's and get their process-local element index, determine the sample mesh element,
484  // and then get the sample mesh DOF from the sample mesh element DOF's. This goes into s2sp.
485  // Note that two sample DOF's could be mapped to the same DOF in sample_mesh, which is fine. This may happen if a DOF is shared by two processes.
486  // The DEIM implementation only sees vectors on each process, without knowledge that the application may identify two DOF's.
487  // Fortunately, DEIM will not select two or more indices on different processes that are equivalent as DOF's, because the corresponding rows
488  // of the basis matrix would then be identical, and then the corresponding entries of the residual used in DEIM would be identical.
489  // Since DEIM does not select the same index twice, it also will not select two indices that have the same row data.
490  // Thus s2sp will not map two indices selected by DEIM to the same DOF in sample_mesh, but it may map two original mesh DOF's that are in
491  // the sample_mesh stencil to the same sample_mesh DOF, which should not cause any problems.
492 
493  int mySampleDofOffset = 0;
494  int os[2] = {0, 0};
495  for (int p=0; p<myid; ++p)
496  {
497  mySampleDofOffset += local_num_sample_dofs[p];
498  for (int i=0; i<2; ++i)
499  os[i] += local_num_sample_dofs_sub[i][p];
500  }
501 
502  // Gather all the sample DOF to element and element DOF indices.
503  vector<int> mySampleToElement(2*local_num_sample_dofs[myid]);
504 
505  mySampleToElement.assign(mySampleToElement.size(), -1); // Initialize with invalid values, to verify later that everything was set.
506 
507  for (int s=0; s<2; ++s) // Loop over subspaces
508  {
509  for (int i=0; i<local_num_sample_dofs_sub[s][myid]; ++i)
510  {
511  const int sdi = sample_dofs_sub_to_sample_dofs[s][os[s] + i] - mySampleDofOffset;
512  mySampleToElement[2*sdi] = localSampleDofsToElem_sub[s][i];
513  mySampleToElement[(2*sdi)+1] = localSampleDofsToElemDof_sub[s][i];
514  }
515  }
516 
517  #ifndef FULL_DOF_STENCIL
518  for (int i=0; i<mySampleToElement.size(); ++i)
519  {
520  MFEM_VERIFY(mySampleToElement[i] >= 0, "");
521  }
522  #endif
523 
524  int* cts = new int [num_procs];
525  int* offsets = new int [num_procs];
526 
527  offsets[0] = 0;
528  cts[0] = 2*local_num_sample_dofs[0];
529  for (int i = 1; i < num_procs; ++i) {
530  cts[i] = 2*local_num_sample_dofs[i];
531  offsets[i] = offsets[i-1] + cts[i-1];
532  }
533 
534  // TODO: replace Allgatherv with just a gather to root?
535 
536  vector<int> sampleToElement(2*global_num_sample_dofs);

883  void GatherDistributedMatrixRows(const CAROM::Matrix& BR, const CAROM::Matrix& BW, const int rrdim, const int rwdim,
884  #ifdef FULL_DOF_STENCIL
885  const int NR, ParFiniteElementSpace& fespaceR, ParFiniteElementSpace& fespaceW,
886  #endif
887  const vector<int>& st2sp, const vector<int>& sprows,
888  const vector<int>& all_sprows, CAROM::Matrix& BRsp, CAROM::Matrix& BWsp)
889  {
890  // Create B sample+ matrix Bsp (B^s+)
891  // On P0 get number of rows each processor contributes to Bsp.
892 
893  int num_procs, myid;
894  MPI_Comm_size(MPI_COMM_WORLD, &num_procs);
895  MPI_Comm_rank(MPI_COMM_WORLD, &myid);
896 
897  //const int rrdim = BR.numColumns();
898  //const int rwdim = BW.numColumns();
899 
900  const int num_sprows = static_cast<int>(sprows.size());
901 
902  #ifdef FULL_DOF_STENCIL
903  int num_sprows_true = 0;
904  for (auto i : sprows)
905  {
906  if ((i < NR && fespaceR.GetLocalTDofNumber(i) >= 0) || (i >= NR && fespaceW.GetLocalTDofNumber(i - NR) >= 0))
907  {
908  num_sprows_true++;
909  }
910  }
911 
912  std::vector<int> sprows_true(num_sprows_true);
913  {
914  int tcnt = 0;
915  for (int j=0; j<sprows.size(); ++j)
916  {
917  const int i = sprows[j];
918  if ((i < NR && fespaceR.GetLocalTDofNumber(i) >= 0) || (i >= NR && fespaceW.GetLocalTDofNumber(i - NR) >= 0))
919  {
920  sprows_true[tcnt] = j;
921  tcnt++;
922  }
923  }
924 
925  MFEM_VERIFY(tcnt == num_sprows_true, "");
926  }
927  #else
928  const int NR = BR.numRows();
929  #endif
930 
931  int* cts = new int [num_procs];
932  int* allNR = new int [num_procs];
933  #ifdef FULL_DOF_STENCIL
934  MPI_Allgather(&num_sprows_true, 1, MPI_INT, cts, 1, MPI_INT, MPI_COMM_WORLD);
935  int* allNSP = new int [num_procs];
936  MPI_Allgather(&num_sprows, 1, MPI_INT, allNSP, 1, MPI_INT, MPI_COMM_WORLD);
937  int* offsetSP = new int [num_procs];
938  offsetSP[0] = 0;
939  for (int i = 1; i < num_procs; ++i) {
940  offsetSP[i] = offsetSP[i-1] + allNSP[i-1];
941  }
942  #else
943  MPI_Allgather(&num_sprows, 1, MPI_INT, cts, 1, MPI_INT, MPI_COMM_WORLD);
944  #endif
945  MPI_Allgather(&NR, 1, MPI_INT, allNR, 1, MPI_INT, MPI_COMM_WORLD);
946  int* offsets = new int [num_procs];
947  offsets[0] = 0;
948  for (int i = 1; i < num_procs; ++i) {
949  offsets[i] = offsets[i-1] + cts[i-1];
950  }
951 
952  #ifdef FULL_DOF_STENCIL
953  const int total_num_sprows_true = offsets[num_procs-1] + cts[num_procs-1];
954  std::vector<int> all_sprows_true(total_num_sprows_true);
955 
956  MPI_Allgatherv(sprows_true.data(), num_sprows_true,
957  MPI_INT, all_sprows_true.data(), cts,
958  offsets, MPI_INT, MPI_COMM_WORLD);
959  #else
960  MFEM_VERIFY(offsets[num_procs-1] + cts[num_procs-1] == all_sprows.size(), "");
961  #endif
962 
963  if (myid == 0) {
964  const int NRsp = BRsp.numRows();
965 
966  for (int i = 0; i < num_sprows; i++) {
967  int row = sprows[i];
968  if (row < NR)
969  {
970  #ifdef FULL_DOF_STENCIL
971  const int ltdof = fespaceR.GetLocalTDofNumber(row);
972  if (ltdof >= 0 && st2sp[i] >= 0)
973  {
974  MFEM_VERIFY(st2sp[i] < NRsp, "");
975 
976  for (int j = 0; j < rrdim; ++j)
977  BRsp(st2sp[i], j) = BR(ltdof, j);
978  }
979  #else
980  for (int j = 0; j < rrdim; ++j)
981  BRsp(st2sp[i], j) = BR(row, j);
982  #endif
983  }
984  else
985  {
986  #ifdef FULL_DOF_STENCIL
987  const int ltdof = fespaceW.GetLocalTDofNumber(row - NR);
988  if (ltdof >= 0 && st2sp[i] >= 0)
989  {
990  MFEM_VERIFY(st2sp[i] - NRsp >= 0 && st2sp[i] - NRsp < BWsp.numRows(), "");
991 
992  for (int j = 0; j < rwdim; ++j)
993  BWsp(st2sp[i] - NRsp, j) = BW(ltdof, j);
994  }
995  #else
996  for (int j = 0; j < rwdim; ++j)
997  BWsp(st2sp[i] - NRsp, j) = BW(row - NR, j);
998  #endif
999  }
1000  }
1001 
1002  #ifdef FULL_DOF_STENCIL
1003  int Bsp_row = num_sprows_true;
1004  #else
1005  int Bsp_row = num_sprows;
1006  #endif
1007  MPI_Status status;
1008  for (int i = 1; i < num_procs; ++i) {
1009  for (int j = 0; j < cts[i]; ++j) {
1010  #ifdef FULL_DOF_STENCIL
1011  const int sti = offsetSP[i] + all_sprows_true[Bsp_row];
1012  if (all_sprows[sti] < allNR[i])
1013  MPI_Recv(&BRsp(st2sp[sti], 0), rrdim, MPI_DOUBLE,
1014  i, offsets[i]+j, MPI_COMM_WORLD, &status); // Note that this may redundantly overwrite some rows corresponding to shared DOF's.
1015  else
1016  MPI_Recv(&BWsp(st2sp[sti] - NRsp, 0), rwdim, MPI_DOUBLE,
1017  i, offsets[i]+j, MPI_COMM_WORLD, &status); // Note that this may redundantly overwrite some rows corresponding to shared DOF's.
1018  #else
1019  if (all_sprows[Bsp_row] < allNR[i])
1020  MPI_Recv(&BRsp(st2sp[Bsp_row], 0), rrdim, MPI_DOUBLE,
1021  i, offsets[i]+j, MPI_COMM_WORLD, &status); // Note that this may redundantly overwrite some rows corresponding to shared DOF's.
1022  else
1023  MPI_Recv(&BWsp(st2sp[Bsp_row] - NRsp, 0), rwdim, MPI_DOUBLE,
1024  i, offsets[i]+j, MPI_COMM_WORLD, &status); // Note that this may redundantly overwrite some rows corresponding to shared DOF's.
1025  #endif
1026  ++Bsp_row;
1027  }
1028  }
1029 
1030  #ifdef FULL_DOF_STENCIL
1031  MFEM_VERIFY(Bsp_row == all_sprows_true.size(), "");
1032  #else
1033  MFEM_VERIFY(Bsp_row == all_sprows.size(), "");
1034  #endif
1035  }
